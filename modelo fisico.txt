CREATE TABLE Pessoa(
	id INT PRIMARY KEY,
	nome varchar(50),
	data_nascimento date,
	email varchar(50),
	senha int
);

CREATE TABLE Projeto(
	codigo INT PRIMARY KEY,
	nome varchar(50),
	data_entrega date
);

CREATE TABLE Estado(
	codigo_tarefa INT PRIMARY KEY,
	todo BIT NOT NULL,
	doing BIT NOT NULL,
	done BIT NOT NULL
);

CREATE TABLE Tarefa(
	codigo integer PRIMARY KEY,
	nome varchar(50),
	data_limite date,
	descricao varchar(300)
);

CREATE TABLE Prioridade(
	codigo_tarefa integer PRIMARY KEY,
	nivel1 BIT NOT NULL,
	nivel2 BIT NOT NULL,
	nivel3 BIT NOT NULL

);

CREATE TABLE Cargo(
	codigo_projeto INT PRIMARY KEY,
	adm int,
	cargo1 int,
	cargo2 int,
	cargo3 int,
	cargo4 int
);
